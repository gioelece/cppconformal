87  MatrixXd regression_vector(n + num_threads, d);
88  regression_matrix << X, MatrixXd::Zero(num_threads, p);
89  regression_vector << Y, MatrixXd::Zero(num_threads, d);
90 
91  std::default_random_engine generator;

93  double tie_breaking = uniform(generator);
94

99  reduction_matrix.reserve(VectorXi::Ones(n + num_threads));
100  for (int i = 0; i < n; i++) {
101  reduction_matrix.insert(i, i) = 1;
102  }
103  reduction_matrix.insert(n, n + this_thread) = 1;
104 
105  #pragma omp for collapse(2)
106  for(int i = 0; i < n0; i++) {
107  for (int j = 0; j < grid.get_size(); j++) {
108  VectorXd y0 = grid.get_point(j);
109  regression_matrix.row(n + this_thread) = Xhat.row(i);
110  regression_vector.row(n + this_thread) = y0;
111

113  auto thread_regression_matrix = reduction_matrix * regression_matrix;
114  auto thread_regression_vector = reduction_matrix * regression_vector;
115 
116  model.fit(thread_regression_matrix, thread_regression_vector);
117  MatrixXd fitted_values = model.predict(thread_regression_matrix);
118  ArrayXd residuals = (thread_regression_vector - fitted_values).rowwise().norm().array();

120  p_values(i, j) = (
121  (residuals > residuals(n)).count()
122  + tie_breaking * (residuals == residuals(n)).count()
123  ) / (n+1.0);
124  }
125  }
